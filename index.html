Tenga en cuenta que todos los valores “n” o “r” son proporcionados por el usuario y utilice las siguientes formulas:
• factorial de un número es: 𝑛! = 𝑛 ∗ (𝑛 − 1) ∗ (𝑛 − 2) ∗ … ∗ 2 ∗ 1
• combinaciones sin repetición es: 𝐶𝑛,𝑟 = 𝑛! / 𝑟!(𝑛−𝑟)!
• combinaciones con repetición es: 𝐶𝑅𝑛,𝑟 = (𝑛+𝑟−1)! / 𝑟!(𝑛−1)! 


// Factorial

static int fact(int n)
{
    if (n == 0 || n == 1)
        return 1;
    else if (n < 0)
        return -1;
    return n * fact(n - 1);
}

// Conbinaciones sin repeticion

static int convsin(int n, int r)
{
    int convinacion = 0;
    convinacion = fact(n) / (fact(r) * fact(n - r));
    return convinacion;
}

// Conbincaiones con repeticiones

static int convcon(int n, int r)
{
    int convinacion = 0;
    convinacion = fact(n + r - 1) / (fact(r) * fact(n - 1));
    return convinacion;
}

static int[] ArregloAscendente(int[] datos)
{
    int auxiliar;
    for (int i = 0; i < datos.Length; i++)
    {
        for (int j = i; j < datos.Length; j++)
        {
            if (datos[j] < datos[i])
            {
                auxiliar = datos[j];
                datos[j] = datos[i];
                datos[i] = auxiliar;
            }
        }
    }
    return datos;
}
