// Factorial

static int fact(int n)
{
    if (n == 0 || n == 1)
        return 1;
    else if (n < 0)
        return -1;
    return n * fact(n - 1);
}

// Conbinaciones sin repeticion

static int convsin(int n, int r)
{
    int convinacion = 0;
    convinacion = fact(n) / (fact(r) * fact(n - r));
    return convinacion;
}

// Conbincaiones con repeticiones

static int convcon(int n, int r)
{
    int convinacion = 0;
    convinacion = fact(n + r - 1) / (fact(r) * fact(n - 1));
    return convinacion;
}
