static void fibonacci(int n)
{
    int a = 0, b = 1, f = 1;
    for (int i = 1; i <= n; i++)
    {
         Console.WriteLine("{0} : {1}", i, f);
         f = a + b;
         a = b;
         b = f;
    }
}

// Factorial y Combinaciones
static int Menu(string mensaje, int inicial, int final)
{
    int opcion;
    do
    {
         Console.ForegroundColor = ConsoleColor.Yellow;
         Console.WriteLine("############ MENU ############");
         Console.WriteLine(mensaje);
         Console.WriteLine("##############################");
         Console.WriteLine("Introduce una opcion entre el {0} al {1}?", inicial, final);
         Console.ForegroundColor = ConsoleColor.Green;
         opcion = int.Parse(Console.ReadLine());
         if (opcion < inicial || opcion > final)
         {
              Console.WriteLine("La opcion debe estar entre {0} al {1}...", inicial, final);
              Console.ReadKey();
              Console.Clear();
         }
         else
              break;
    } while (true);
    return opcion;
}

static long Factorial(double numero = 1)
{
    long resultado = 1;
    if (numero < 0)
    {
         Console.WriteLine("No existe faltorial para numeros negativos");          resultado = -1;
    }
    else if (numero > 1)
    {
         for (int i = 1; i <= numero; i++)
              resultado *= i; // resultado = resultado * i
    }
    return resultado;
}

static void Numero(out double n)
{
    Console.Write("Introduce numero? ");
    n = double.Parse(Console.ReadLine());
}

static long Factorial(double n = 1)
{
    long resultado = 1;
    if (n < 0)
    {
        Console.WriteLine("No existe faltorial para numeros negativos");
        resultado = -1;
    }
    else if (n > 1)
    {
        for (int i = 1; i <= n; i++)
            resultado *= i; // resultado = resultado * i
    }
    return resultado;
}

do

    double opcion, n = 0;

    if (opcion == 1)
        Console.WriteLine("Opcion 1");
    else if (opcion == 2)
