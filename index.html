static long Factorial(double numero = 1)
{
    long resultado = 1;
    if (numero < 0)
    {
         Console.WriteLine("No existe faltorial para numeros negativos");          resultado = -1;
    }
    else if (numero > 1)
    {
         for (int i = 1; i <= numero; i++)
              resultado *= i; // resultado = resultado * i
    }
    return resultado;
}

static int factorial (int n)
{
    if (n == 0 || n == 1)
        return 1;
    else if (n < 0)
        return -1;
    return n * factorial(n - 1);
}

static int CombinacionSin (int n, int r)
{
    int combinacion = 0;
    combinacion = factorial(n) / (factorial(r) * factorial(n - r));
    return combinacion;
}

static int CombinacionCon(int n, int r)
{
    int combinacion = 0;
    combinacion = factorial(n + r - 1) / (factorial(r) * factorial(n - 1));
    return combinacion;
}

static void OcurrenciasTexto(bool indice = true)

static void CantidadOpcion3()
{
    string texto;
    int canVocales = 0, canEspeciales = 0, cantEspacios = 0;
    Console.WriteLine("Introduce un texto cualquiera? ");
    texto = Console.ReadLine();
    foreach (char c in texto.ToUpper())
    {
        if (char.IsWhiteSpace(c))
            cantEspacios++;
        else if (c == '$' || c == '%' || c == '^' || c == '&' || c == '#')
            canEspeciales++;
        else if (c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U')
            canVocales++;
    }
    Console.WriteLine("Cantidad de vocales: {0}\nCantidad $%^&#: {1}\nCantidad espacios: {2}", canVocales, canEspeciales, cantEspacios);
}

static void OrdenDescendente(int[] arreglo)
{
    Array.Sort(arreglo);
    Array.Reverse(arreglo);
    Console.WriteLine("Orden descendente: ");
    foreach (int i in arreglo)
        Console.Write(i + " ");
}
static void OrdenAscendente(int[] arreglo)
{
    int aux = 0;
    for (int i = 0; i < arreglo.Length; i++)
    {
        for (int j = i + 1; j < arreglo.Length; j++)
        {
            if (arreglo[j] < arreglo[i])
            {
                aux = arreglo[i];
                arreglo[i] = arreglo[j];
                arreglo[j] = aux;
            }
        }
    }
    Console.WriteLine("Ordenado Ascendentemente:");
    foreach (int a in arreglo)
        Console.Write(a + " ");

}

static int[] GenerarArreglo(int cantidad, int inicio = 0, int fin = 100)
{
    int[] datos = new int[cantidad];
    Random valor = new Random();
    for (int i = 0; i < cantidad; i++)
    {
        datos[i] = valor.Next(inicio, fin + 1);
        Console.Write(datos[i] + " ");
    }
    return datos;
}
